import React, { useState, useRef, useMemo } from 'react';
import { Target, XCircle, CheckCircle2, RotateCcw, Activity, Undo2, BarChart2, Smartphone, Monitor, RefreshCw, Hand, Magnet, ArrowUpCircle, AlertOctagon, MousePointer2 } from 'lucide-react';

export default function App() {
  const [mode, setMode] = useState('Standard');
  const [events, setEvents] = useState([]); 
  const [activeTeam, setActiveTeam] = useState('A');
  const [activeTool, setActiveTool] = useState('shot'); 
  const [teamADirection, setTeamADirection] = useState('left'); // 'left' or 'right'
  const [ssnSuperShotActive, setSsnSuperShotActive] = useState(false);
  const [fast5PowerPlay, setFast5PowerPlay] = useState(false);
  
  const [isVertical, setIsVertical] = useState(false);
  const [hoverInfo, setHoverInfo] = useState(null);
  const [dragState, setDragState] = useState(null);
  const [pendingStat, setPendingStat] = useState(null);

  const svgRef = useRef(null);

  // Court dimensions in generic units (1 unit = 0.1m, so 305 x 152.5)
  const COURT_WIDTH = 305;
  const COURT_HEIGHT = 152.5;
  const THIRD_WIDTH = 101.66;
  const POST_Y = COURT_HEIGHT / 2;

  const toDataCoords = (visX, visY) => ({
    x: isVertical ? visY : visX,
    y: isVertical ? COURT_HEIGHT - visX : visY
  });

  const toVisCoords = (dataX, dataY) => ({
    x: isVertical ? COURT_HEIGHT - dataY : dataX,
    y: isVertical ? dataX : dataY
  });

  // SMART POSITIONAL FILTERING based on Direction of Play
  const getValidPositionsForTeam = (x, y, team, currentMode, directionA) => {
    const isLeftThird = x <= THIRD_WIDTH;
    const isRightThird = x >= COURT_WIDTH - THIRD_WIDTH;
    const isCenterThird = !isLeftThird && !isRightThird;

    const isLeftCircle = Math.sqrt(Math.pow(x - 0, 2) + Math.pow(y - POST_Y, 2)) <= 49 && isLeftThird;
    const isRightCircle = Math.sqrt(Math.pow(x - COURT_WIDTH, 2) + Math.pow(y - POST_Y, 2)) <= 49 && isRightThird;

    const attackingLeft = (team === 'A' && directionA === 'left') || (team === 'B' && directionA === 'right');

    let validPos = [];
    
    if (isCenterThird) {
      validPos = ['GA', 'WA', 'C', 'WD', 'GD'];
    } else if (isLeftCircle) {
      // If team is attacking this circle, only GS/GA. If defending, only GK/GD.
      validPos = attackingLeft ? ['GS', 'GA'] : ['GK', 'GD'];
    } else if (isRightCircle) {
      validPos = !attackingLeft ? ['GS', 'GA'] : ['GK', 'GD'];
    } else if (isLeftThird) {
      // In the third, but outside the circle
      validPos = attackingLeft ? ['GS', 'GA', 'WA', 'C'] : ['GK', 'GD', 'WD', 'C'];
    } else if (isRightThird) {
      validPos = !attackingLeft ? ['GS', 'GA', 'WA', 'C'] : ['GK', 'GD', 'WD', 'C'];
    }

    // Fast5 modifier: WA and WD do not exist
    if (currentMode === 'Fast5') {
      validPos = validPos.filter(pos => pos !== 'WA' && pos !== 'WD');
    }

    return validPos;
  };

  const getShotDetails = (x, y, team, currentMode, directionA) => {
    const isLeftThird = x <= THIRD_WIDTH;
    const isRightThird = x >= COURT_WIDTH - THIRD_WIDTH;
    const attackingLeft = (team === 'A' && directionA === 'left') || (team === 'B' && directionA === 'right');
    
    // Cannot shoot from wrong end or center third
    if (attackingLeft && !isLeftThird) return { valid: false, points: 0, zoneName: 'Invalid Scoring End' };
    if (!attackingLeft && !isRightThird) return { valid: false, points: 0, zoneName: 'Invalid Scoring End' };

    const postX = attackingLeft ? 0 : COURT_WIDTH;
    const distance = Math.sqrt(Math.pow(x - postX, 2) + Math.pow(y - POST_Y, 2));

    let points = 0;
    let zoneName = '';

    switch (currentMode) {
      case 'Standard':
        if (distance <= 49) { points = 1; zoneName = 'Goal Circle'; } 
        else { zoneName = 'Outside Circle (Invalid)'; }
        break;
      case 'SSN':
        if (distance <= 30) { points = 1; zoneName = 'Inner Circle'; } 
        else if (distance <= 49) { points = ssnSuperShotActive ? 2 : 1; zoneName = ssnSuperShotActive ? 'Super Shot Zone' : 'Outer Circle'; } 
        else { zoneName = 'Outside Circle (Invalid)'; }
        break;
      case 'Fast5':
        if (distance <= 30) { points = 1; zoneName = 'Inner Circle'; } 
        else if (distance <= 49) { points = 2; zoneName = 'Outer Circle'; } 
        else { points = 3; zoneName = 'Super Shot Zone'; }
        if (fast5PowerPlay) points *= 2;
        break;
      case 'Indoor':
        if (distance <= 49) { points = 1; zoneName = 'Goal Circle'; } 
        else { points = 2; zoneName = '2-Point Zone'; }
        break;
      default:
        points = 0;
    }

    return { valid: points > 0 || (currentMode === 'Indoor'), points, zoneName };
  };

  const handlePointerDown = (e) => {
    if (!svgRef.current) return;
    const rect = svgRef.current.getBoundingClientRect();
    const vWidth = isVertical ? COURT_HEIGHT : COURT_WIDTH;
    const vHeight = isVertical ? COURT_WIDTH : COURT_HEIGHT;
    const scaleX = vWidth / rect.width;
    const scaleY = vHeight / rect.height;
    
    const visX = (e.clientX - rect.left) * scaleX;
    const visY = (e.clientY - rect.top) * scaleY;

    const dataCoords = toDataCoords(visX, visY);

    if (activeTool === 'shot') {
      const details = getShotDetails(dataCoords.x, dataCoords.y, activeTeam, mode, teamADirection);
      if (details.valid) {
        e.currentTarget.setPointerCapture(e.pointerId);
        setDragState({
          phase: 'position', 
          visStartX: visX, visStartY: visY,
          visCurrentX: visX, visCurrentY: visY,
          dataStartX: dataCoords.x, dataStartY: dataCoords.y,
          selectedPosition: null, selectedResult: null,
          visPhase2StartX: null, visPhase2StartY: null,
          details
        });
      }
    } else {
      setPendingStat({
        visX, visY,
        dataX: dataCoords.x,
        dataY: dataCoords.y,
        team: activeTeam,
        type: activeTool,
        mode,
        directionA: teamADirection
      });
    }
  };

  const handlePointerMove = (e) => {
    if (!svgRef.current) return;
    const rect = svgRef.current.getBoundingClientRect();
    const vWidth = isVertical ? COURT_HEIGHT : COURT_WIDTH;
    const vHeight = isVertical ? COURT_WIDTH : COURT_HEIGHT;
    const scaleX = vWidth / rect.width;
    const scaleY = vHeight / rect.height;
    
    const visX = (e.clientX - rect.left) * scaleX;
    const visY = (e.clientY - rect.top) * scaleY;

    if (dragState && activeTool === 'shot') {
      if (dragState.phase === 'position') {
        const dx = visX - dragState.visStartX;
        if (Math.abs(dx) > 12) {
          setDragState(prev => ({
            ...prev,
            visCurrentX: visX, visCurrentY: visY,
            phase: 'result', 
            selectedPosition: dx < 0 ? 'GS' : 'GA',
            visPhase2StartX: visX, visPhase2StartY: visY
          }));
        } else {
          setDragState(prev => ({ ...prev, visCurrentX: visX, visCurrentY: visY }));
        }
      } else if (dragState.phase === 'result' || dragState.phase === 'ready') {
        const dy = visY - dragState.visPhase2StartY;
        let newResult = null;
        let newPhase = 'result';

        if (Math.abs(dy) > 12) {
          newResult = dy < 0 ? 'made' : 'missed';
          newPhase = 'ready'; 
        }

        setDragState(prev => ({
          ...prev,
          visCurrentX: visX, visCurrentY: visY,
          selectedResult: newResult,
          phase: newPhase
        }));
      }
    } else {
      if (pendingStat) {
        setHoverInfo(null);
        return;
      }

      const dataCoords = toDataCoords(visX, visY);
      if (activeTool === 'shot') {
        const details = getShotDetails(dataCoords.x, dataCoords.y, activeTeam, mode, teamADirection);
        setHoverInfo({ x: dataCoords.x, y: dataCoords.y, type: 'shot', ...details });
      } else {
        setHoverInfo({ x: dataCoords.x, y: dataCoords.y, type: activeTool, valid: true });
      }
    }
  };

  const handlePointerUp = (e) => {
    if (dragState && activeTool === 'shot') {
      e.currentTarget.releasePointerCapture(e.pointerId);
      
      if (dragState.selectedPosition && dragState.selectedResult) {
        const newShot = {
          id: Date.now(),
          x: dragState.dataStartX, 
          y: dragState.dataStartY,
          team: activeTeam,
          type: 'shot',
          position: dragState.selectedPosition,
          result: dragState.selectedResult,
          points: dragState.selectedResult === 'made' ? dragState.details.points : 0,
          potentialPoints: dragState.details.points,
          mode
        };
        setEvents([...events, newShot]);
      }

      setDragState(null);
      setHoverInfo(null);
    }
  };

  const handlePointerLeave = () => {
    if (!dragState) setHoverInfo(null);
  };

  const handleRadialPositionSelect = (e, position) => {
    e.stopPropagation(); 
    if (!pendingStat) return;

    const newEvent = {
      id: Date.now(),
      x: pendingStat.dataX,
      y: pendingStat.dataY,
      team: pendingStat.team,
      type: pendingStat.type,
      position: position,
      mode: pendingStat.mode
    };
    
    setEvents([...events, newEvent]);
    setPendingStat(null); 
  };

  const stats = useMemo(() => {
    const defaultStats = { score: 0, attempts: 0, makes: 0, rebounds: 0, turnovers: 0, intercepts: 0, deflections: 0, penalties: 0 };
    const result = { A: { ...defaultStats }, B: { ...defaultStats } };

    events.forEach(ev => {
      if (ev.type === 'shot') {
        result[ev.team].attempts += 1;
        if (ev.result === 'made') {
          result[ev.team].makes += 1;
          result[ev.team].score += ev.points;
        }
      } else if (ev.type === 'rebound') result[ev.team].rebounds += 1;
      else if (ev.type === 'turnover') result[ev.team].turnovers += 1;
      else if (ev.type === 'intercept') result[ev.team].intercepts += 1;
      else if (ev.type === 'deflection') result[ev.team].deflections += 1;
      else if (ev.type === 'penalty') result[ev.team].penalties += 1;
    });

    result.A.pct = result.A.attempts > 0 ? Math.round((result.A.makes / result.A.attempts) * 100) : 0;
    result.B.pct = result.B.attempts > 0 ? Math.round((result.B.makes / result.B.attempts) * 100) : 0;

    return result;
  }, [events]);

  const clearCourt = () => { setEvents([]); setPendingStat(null); };
  const undoLastEvent = () => { setEvents(events.slice(0, -1)); setPendingStat(null); };

  // Visual Setup
  const courtWoodColor = "#dfb17e";
  const courtLineColor = "#ffffff";
  const postColor = "#ef4444"; 

  const renderGoalCircle = (cx, cy, isLeft) => {
    const sweep = isLeft ? 1 : 0;
    return <path d={`M ${cx},${cy - 49} A 49,49 0 0,${sweep} ${cx},${cy + 49}`} fill="rgba(255, 255, 255, 0.15)" stroke={courtLineColor} strokeWidth="1.5" />;
  };

  const renderFast5Zones = (cx, cy, isLeft) => {
    const sweep = isLeft ? 1 : 0;
    return (
      <>
        <path d={`M ${cx},${cy - 49} A 49,49 0 0,${sweep} ${cx},${cy + 49}`} fill="rgba(234, 179, 8, 0.25)" stroke={courtLineColor} strokeWidth="1.5"/>
        <path d={`M ${cx},${cy - 30} A 30,30 0 0,${sweep} ${cx},${cy + 30}`} fill="rgba(59, 130, 246, 0.35)" stroke={courtLineColor} strokeWidth="1.5"/>
      </>
    );
  };

  const renderSSNZones = (cx, cy, isLeft) => {
    const sweep = isLeft ? 1 : 0;
    return (
      <>
        <path d={`M ${cx},${cy - 49} A 49,49 0 0,${sweep} ${cx},${cy + 49}`} fill={ssnSuperShotActive ? "rgba(239, 68, 68, 0.3)" : "rgba(255, 255, 255, 0.15)"} stroke={courtLineColor} strokeWidth="1.5"/>
        <path d={`M ${cx},${cy - 30} A 30,30 0 0,${sweep} ${cx},${cy + 30}`} fill="rgba(255, 255, 255, 0.2)" stroke={courtLineColor} strokeWidth="1.5" strokeDasharray="4,2"/>
      </>
    );
  };

  const renderEventMarker = (ev, vX, vY) => {
    const teamColor = ev.team === 'A' ? '#4f46e5' : '#e11d48';
    
    if (ev.type === 'shot') {
      return (
        <g>
          <circle cx={vX} cy={vY} r="3.5" fill={teamColor} stroke={ev.result === 'made' ? '#10b981' : '#f43f5e'} strokeWidth="2" />
          {ev.result === 'missed' && (
            <>
              <line x1={vX - 2.5} y1={vY - 2.5} x2={vX + 2.5} y2={vY + 2.5} stroke="white" strokeWidth="1.5" strokeLinecap="round" />
              <line x1={vX + 2.5} y1={vY - 2.5} x2={vX - 2.5} y2={vY + 2.5} stroke="white" strokeWidth="1.5" strokeLinecap="round" />
            </>
          )}
          <text x={vX} y={vY - 5} fontSize="3.5" fontWeight="bold" fill="white" textAnchor="middle" opacity="0.8">{ev.position}</text>
        </g>
      );
    }
    
    let color = '#333';
    let label = '';
    
    switch (ev.type) {
      case 'rebound': color = '#8b5cf6'; label = 'REB'; break; 
      case 'turnover': color = '#f97316'; label = 'TO'; break;  
      case 'intercept': color = '#06b6d4'; label = 'INT'; break; 
      case 'deflection': color = '#14b8a6'; label = 'DEF'; break; 
      case 'penalty': color = '#eab308'; label = 'PEN'; break;  
    }

    return (
      <g>
        <rect x={vX - 3.5} y={vY - 3.5} width="7" height="7" fill={color} stroke={teamColor} strokeWidth="1" rx="1.5" />
        <text x={vX} y={vY + 1.2} fontSize="2.5" fontWeight="black" fill="white" textAnchor="middle">{label}</text>
        {ev.position && <text x={vX + 5} y={vY + 1} fontSize="2.5" fontWeight="bold" fill={teamColor} stroke="white" strokeWidth="0.2" textAnchor="start">{ev.position}</text>}
      </g>
    );
  };

  return (
    <div className="min-h-screen bg-slate-100 text-slate-800 p-4 md:p-8 font-sans flex flex-col">
      {/* Header */}
      <header className="flex flex-col md:flex-row justify-between items-center mb-6 gap-4 bg-white p-5 rounded-2xl shadow-sm border border-slate-200">
        <div className="flex items-center gap-3">
          <div className="p-3 bg-gradient-to-br from-blue-600 to-indigo-700 rounded-xl shadow-inner">
            <Activity className="text-white w-6 h-6" />
          </div>
          <div>
            <h1 className="text-2xl font-extrabold text-slate-900 tracking-tight">Netball Analytics <span className="text-blue-500 font-black">V2</span></h1>
            <p className="text-slate-500 text-sm font-semibold">Directional & Positional Tracking</p>
          </div>
        </div>

        <div className="flex flex-col sm:flex-row items-center gap-4">
          <div className="flex flex-wrap gap-2 p-1 bg-slate-100 rounded-full border border-slate-200">
            {['Standard', 'SSN', 'Fast5', 'Indoor'].map((m) => (
              <button
                key={m}
                onClick={() => { setMode(m); clearCourt(); }}
                className={`px-5 py-2 rounded-full text-sm font-bold transition-all ${
                  mode === m ? 'bg-white text-blue-700 shadow-sm border border-slate-200' : 'text-slate-500 hover:text-slate-800 hover:bg-slate-200/50'
                }`}
              >
                {m}
              </button>
            ))}
          </div>
          <button
            onClick={() => setIsVertical(!isVertical)}
            className="flex items-center gap-2 px-4 py-2 rounded-full text-sm font-bold bg-slate-100 text-slate-600 hover:bg-slate-200 transition-all border border-slate-200"
          >
            {isVertical ? <><Smartphone className="w-4 h-4"/> Vertical</> : <><Monitor className="w-4 h-4"/> Horizontal</>}
          </button>
        </div>
      </header>

      <div className="flex flex-col lg:flex-row gap-6 flex-1">
        
        {/* Left Column: Controls */}
        <div className="w-full lg:w-72 flex flex-col gap-6">
          <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-200">
            <h2 className="text-xs font-black text-slate-400 uppercase tracking-widest mb-5">Match State</h2>
            
            <div className="space-y-6">
              <div>
                <label className="text-sm font-bold text-slate-700 block mb-3">Possession / Active Team</label>
                <div className="flex gap-2">
                  <button onClick={() => {setActiveTeam('A'); setPendingStat(null);}} className={`flex-1 py-3 rounded-xl font-bold border-2 transition-all shadow-sm ${activeTeam === 'A' ? 'border-indigo-600 bg-indigo-600 text-white' : 'border-slate-100 bg-slate-50 text-slate-500 hover:bg-slate-100'}`}>Team A (Blue)</button>
                  <button onClick={() => {setActiveTeam('B'); setPendingStat(null);}} className={`flex-1 py-3 rounded-xl font-bold border-2 transition-all shadow-sm ${activeTeam === 'B' ? 'border-rose-600 bg-rose-600 text-white' : 'border-slate-100 bg-slate-50 text-slate-500 hover:bg-slate-100'}`}>Team B (Red)</button>
                </div>
              </div>

              {/* Scoring Direction Toggle */}
              <div className="pt-2 border-t border-slate-100">
                <label className="text-sm font-bold text-slate-700 block mb-3 mt-2">Team A Scoring Direction</label>
                <div className="flex gap-2">
                  <button onClick={() => setTeamADirection('left')} className={`flex-1 py-2 rounded-xl font-bold border-2 transition-all shadow-sm ${teamADirection === 'left' ? 'border-indigo-500 bg-indigo-50 text-indigo-700' : 'border-slate-100 bg-white text-slate-500 hover:bg-slate-50'}`}>⬅ Left Hoop</button>
                  <button onClick={() => setTeamADirection('right')} className={`flex-1 py-2 rounded-xl font-bold border-2 transition-all shadow-sm ${teamADirection === 'right' ? 'border-indigo-500 bg-indigo-50 text-indigo-700' : 'border-slate-100 bg-white text-slate-500 hover:bg-slate-50'}`}>Right Hoop ➡</button>
                </div>
              </div>

              {/* Event Tool Palette */}
              <div className="pt-4 border-t border-slate-100">
                <label className="text-sm font-bold text-slate-700 block mb-3">Action Tool</label>
                <div className="grid grid-cols-2 gap-2">
                  <button onClick={() => {setActiveTool('shot'); setPendingStat(null);}} className={`flex flex-col items-center justify-center p-3 rounded-xl border-2 transition-all ${activeTool === 'shot' ? 'border-blue-500 bg-blue-50 text-blue-700' : 'border-slate-100 bg-white text-slate-500 hover:bg-slate-50'}`}>
                    <MousePointer2 className="w-5 h-5 mb-1" /> <span className="text-[10px] font-bold uppercase tracking-wide">Shot (Drag)</span>
                  </button>
                  <button onClick={() => {setActiveTool('rebound'); setPendingStat(null);}} className={`flex flex-col items-center justify-center p-3 rounded-xl border-2 transition-all ${activeTool === 'rebound' ? 'border-purple-500 bg-purple-50 text-purple-700' : 'border-slate-100 bg-white text-slate-500 hover:bg-slate-50'}`}>
                    <ArrowUpCircle className="w-5 h-5 mb-1" /> <span className="text-[10px] font-bold uppercase tracking-wide">Rebound</span>
                  </button>
                  <button onClick={() => {setActiveTool('turnover'); setPendingStat(null);}} className={`flex flex-col items-center justify-center p-3 rounded-xl border-2 transition-all ${activeTool === 'turnover' ? 'border-orange-500 bg-orange-50 text-orange-700' : 'border-slate-100 bg-white text-slate-500 hover:bg-slate-50'}`}>
                    <RefreshCw className="w-5 h-5 mb-1" /> <span className="text-[10px] font-bold uppercase tracking-wide">Turnover</span>
                  </button>
                  <button onClick={() => {setActiveTool('intercept'); setPendingStat(null);}} className={`flex flex-col items-center justify-center p-3 rounded-xl border-2 transition-all ${activeTool === 'intercept' ? 'border-cyan-500 bg-cyan-50 text-cyan-700' : 'border-slate-100 bg-white text-slate-500 hover:bg-slate-50'}`}>
                    <Magnet className="w-5 h-5 mb-1" /> <span className="text-[10px] font-bold uppercase tracking-wide">Intercept</span>
                  </button>
                  <button onClick={() => {setActiveTool('deflection'); setPendingStat(null);}} className={`flex flex-col items-center justify-center p-3 rounded-xl border-2 transition-all ${activeTool === 'deflection' ? 'border-teal-500 bg-teal-50 text-teal-700' : 'border-slate-100 bg-white text-slate-500 hover:bg-slate-50'}`}>
                    <Hand className="w-5 h-5 mb-1" /> <span className="text-[10px] font-bold uppercase tracking-wide">Deflection</span>
                  </button>
                  <button onClick={() => {setActiveTool('penalty'); setPendingStat(null);}} className={`flex flex-col items-center justify-center p-3 rounded-xl border-2 transition-all ${activeTool === 'penalty' ? 'border-yellow-500 bg-yellow-50 text-yellow-700' : 'border-slate-100 bg-white text-slate-500 hover:bg-slate-50'}`}>
                    <AlertOctagon className="w-5 h-5 mb-1" /> <span className="text-[10px] font-bold uppercase tracking-wide">Penalty</span>
                  </button>
                </div>
              </div>

              {/* Format Specific Toggles */}
              {mode === 'SSN' && (
                <div className="pt-5 border-t border-slate-100">
                  <label className="flex items-center justify-between cursor-pointer group">
                    <span className="text-sm font-bold text-slate-700 group-hover:text-red-600 transition-colors">Super Shot Active (Last 5m)</span>
                    <input type="checkbox" checked={ssnSuperShotActive} onChange={(e) => setSsnSuperShotActive(e.target.checked)} className="w-5 h-5 accent-red-600 cursor-pointer" />
                  </label>
                </div>
              )}

              {mode === 'Fast5' && (
                <div className="pt-5 border-t border-slate-100">
                  <label className="flex items-center justify-between cursor-pointer group">
                    <span className="text-sm font-bold text-slate-700 group-hover:text-yellow-600 transition-colors">Power Play (2x Points)</span>
                    <input type="checkbox" checked={fast5PowerPlay} onChange={(e) => setFast5PowerPlay(e.target.checked)} className="w-5 h-5 accent-yellow-500 cursor-pointer" />
                  </label>
                </div>
              )}
            </div>
          </div>
        </div>

        {/* Center: Court & Scoreboard */}
        <div className="flex-1 flex flex-col items-center min-w-0">
          
          {/* Advanced Scoreboard */}
          <div className="w-full bg-white px-6 py-4 rounded-t-2xl border border-slate-200 border-b-0 shadow-sm relative z-10">
            <div className="flex justify-between items-start">
              {/* Team A Stats */}
              <div className="flex flex-col items-start w-1/3">
                <div className="text-sm font-black text-indigo-600 uppercase tracking-widest mb-1">Team A (Blue)</div>
                <div className="flex items-baseline gap-3">
                  <span className="text-5xl font-black text-slate-900">{stats.A.score}</span>
                  <div className="flex flex-col">
                    <span className="text-xs font-bold text-slate-400">{stats.A.makes}/{stats.A.attempts} FG</span>
                    <span className="text-sm font-bold text-indigo-600">{stats.A.pct}%</span>
                  </div>
                </div>
                <div className="flex flex-wrap gap-2 text-[10px] font-black text-slate-400 mt-2 uppercase">
                  <span title="Rebounds">REB <span className="text-slate-700">{stats.A.rebounds}</span></span>
                  <span title="Turnovers">TO <span className="text-slate-700">{stats.A.turnovers}</span></span>
                  <span title="Intercepts">INT <span className="text-slate-700">{stats.A.intercepts}</span></span>
                  <span title="Deflections">DEF <span className="text-slate-700">{stats.A.deflections}</span></span>
                  <span title="Penalties">PEN <span className="text-slate-700">{stats.A.penalties}</span></span>
                </div>
              </div>
              
              <div className="flex flex-col items-center justify-center w-1/3 mt-2">
                <div className="text-[10px] font-black text-slate-400 bg-slate-100 px-3 py-1 rounded-full mb-2 tracking-widest border border-slate-200">
                  LIVE TRACKER
                </div>
                <div className="text-lg font-black text-slate-800 uppercase">{mode}</div>
              </div>

              {/* Team B Stats */}
              <div className="flex flex-col items-end w-1/3 text-right">
                <div className="text-sm font-black text-rose-600 uppercase tracking-widest mb-1">Team B (Red)</div>
                <div className="flex items-baseline gap-3 flex-row-reverse">
                  <span className="text-5xl font-black text-slate-900">{stats.B.score}</span>
                  <div className="flex flex-col items-end">
                    <span className="text-xs font-bold text-slate-400">{stats.B.makes}/{stats.B.attempts} FG</span>
                    <span className="text-sm font-bold text-rose-600">{stats.B.pct}%</span>
                  </div>
                </div>
                <div className="flex flex-wrap gap-2 text-[10px] font-black text-slate-400 mt-2 uppercase flex-row-reverse">
                  <span title="Rebounds"><span className="text-slate-700">{stats.B.rebounds}</span> REB</span>
                  <span title="Turnovers"><span className="text-slate-700">{stats.B.turnovers}</span> TO</span>
                  <span title="Intercepts"><span className="text-slate-700">{stats.B.intercepts}</span> INT</span>
                  <span title="Deflections"><span className="text-slate-700">{stats.B.deflections}</span> DEF</span>
                  <span title="Penalties"><span className="text-slate-700">{stats.B.penalties}</span> PEN</span>
                </div>
              </div>
            </div>
          </div>

          {/* Interactive Court Area */}
          <div className={`w-full overflow-hidden relative shadow-lg cursor-crosshair border-8 border-white group bg-slate-900 select-none touch-none transition-all duration-300 ${isVertical ? 'aspect-[1/2] max-w-sm sm:max-w-md mx-auto rounded-2xl mt-4' : 'aspect-[2/1] rounded-b-2xl'}`}>
            <svg
              ref={svgRef}
              viewBox={`0 0 ${isVertical ? COURT_HEIGHT : COURT_WIDTH} ${isVertical ? COURT_WIDTH : COURT_HEIGHT}`}
              className="w-full h-full"
              onPointerDown={handlePointerDown}
              onPointerMove={handlePointerMove}
              onPointerUp={handlePointerUp}
              onPointerLeave={handlePointerLeave}
              onPointerCancel={handlePointerUp}
            >
              <g transform={isVertical ? `translate(${COURT_HEIGHT}, 0) rotate(90)` : ""}>
                <rect x="0" y="0" width={COURT_WIDTH} height={COURT_HEIGHT} fill={courtWoodColor} />
                
                {/* Court Direction Typography Identifiers */}
                <g opacity="0.12" style={{ pointerEvents: 'none' }}>
                  <text transform={`translate(28, ${COURT_HEIGHT/2}) rotate(-90)`} fontSize="14" fontWeight="black" fill={teamADirection === 'left' ? '#4f46e5' : '#e11d48'} textAnchor="middle" letterSpacing="4">
                    {teamADirection === 'left' ? 'TEAM A SCORING' : 'TEAM B SCORING'}
                  </text>
                  <text transform={`translate(${COURT_WIDTH - 28}, ${COURT_HEIGHT/2}) rotate(90)`} fontSize="14" fontWeight="black" fill={teamADirection === 'right' ? '#4f46e5' : '#e11d48'} textAnchor="middle" letterSpacing="4">
                    {teamADirection === 'right' ? 'TEAM A SCORING' : 'TEAM B SCORING'}
                  </text>
                </g>

                <line x1={THIRD_WIDTH} y1="0" x2={THIRD_WIDTH} y2={COURT_HEIGHT} stroke={courtLineColor} strokeWidth="1.5" />
                <line x1={COURT_WIDTH - THIRD_WIDTH} y1="0" x2={COURT_WIDTH - THIRD_WIDTH} y2={COURT_HEIGHT} stroke={courtLineColor} strokeWidth="1.5" />
                <circle cx={COURT_WIDTH / 2} cy={POST_Y} r="9" fill="none" stroke={courtLineColor} strokeWidth="1.5" />

                {mode === 'Standard' || mode === 'Indoor' ? (
                  <>{renderGoalCircle(0, POST_Y, true)}{renderGoalCircle(COURT_WIDTH, POST_Y, false)}</>
                ) : mode === 'Fast5' ? (
                  <>{renderFast5Zones(0, POST_Y, true)}{renderFast5Zones(COURT_WIDTH, POST_Y, false)}</>
                ) : mode === 'SSN' ? (
                  <>{renderSSNZones(0, POST_Y, true)}{renderSSNZones(COURT_WIDTH, POST_Y, false)}</>
                ) : null}

                <circle cx={0} cy={POST_Y} r="1.5" fill={postColor} />
                <circle cx={COURT_WIDTH} cy={POST_Y} r="1.5" fill={postColor} />

                {mode === 'Indoor' && (
                  <rect x="2" y="2" width={COURT_WIDTH-4} height={COURT_HEIGHT-4} fill="none" stroke="rgba(255,255,255,0.4)" strokeWidth="2" strokeDasharray="4,4" />
                )}
              </g>

              {/* Render Recorded Events */}
              {events.map((ev) => {
                const { x: vX, y: vY } = toVisCoords(ev.x, ev.y);
                return <g key={ev.id} className="transition-all animate-in fade-in duration-300">{renderEventMarker(ev, vX, vY)}</g>;
              })}

              {/* SMART RADIAL MENU UI (For Rebounds, Turnovers, etc.) */}
              {pendingStat && (
                <g className="transition-all drop-shadow-lg pointer-events-auto">
                  {/* Backdrop */}
                  <circle cx={pendingStat.visX} cy={pendingStat.visY} r="28" fill="rgba(15, 23, 42, 0.75)" stroke="rgba(255,255,255,0.2)" strokeWidth="0.5" />
                  
                  {/* Cancel Button (Center) */}
                  <g transform={`translate(${pendingStat.visX}, ${pendingStat.visY})`} onPointerDown={(e) => { e.stopPropagation(); setPendingStat(null); }} style={{cursor: 'pointer'}}>
                    <circle r="5" fill="#ef4444" stroke="white" strokeWidth="0.5" />
                    <text x="0" y="1.2" fontSize="3" fontWeight="bold" fill="white" textAnchor="middle">✕</text>
                  </g>

                  {/* Dynamically Filtered Positions using strict attacking/defending boundaries */}
                  {getValidPositionsForTeam(pendingStat.dataX, pendingStat.dataY, pendingStat.team, pendingStat.mode, pendingStat.directionA).map((pos, index, arr) => {
                    const angle = (index * (Math.PI * 2)) / arr.length - Math.PI / 2; 
                    const px = pendingStat.visX + 18 * Math.cos(angle);
                    const py = pendingStat.visY + 18 * Math.sin(angle);
                    const isTeamA = activeTeam === 'A';
                    
                    return (
                      <g 
                        key={pos} 
                        transform={`translate(${px}, ${py})`} 
                        onPointerDown={(e) => handleRadialPositionSelect(e, pos)} 
                        style={{cursor: 'pointer'}}
                      >
                        <circle r="5.5" fill={isTeamA ? '#4f46e5' : '#e11d48'} stroke="white" strokeWidth="0.8" />
                        <text x="0" y="1.5" fontSize="3.5" fontWeight="bold" fill="white" textAnchor="middle">{pos}</text>
                      </g>
                    );
                  })}
                </g>
              )}

              {/* 2-STAGE DRAG UI HUD (Only for shots) */}
              {dragState && activeTool === 'shot' && (
                <g className="pointer-events-none transition-all drop-shadow-lg">
                  {dragState.phase !== 'position' && (
                    <g>
                      <circle cx={dragState.visStartX} cy={dragState.visStartY} r="2.5" fill={activeTeam === 'A' ? '#4f46e5' : '#e11d48'} stroke="white" strokeWidth="0.5"/>
                      <text x={dragState.visStartX} y={dragState.visStartY - 4} fontSize="4" fontWeight="bold" fill="white" textAnchor="middle">
                        {dragState.selectedPosition}
                      </text>
                      <line x1={dragState.visStartX} y1={dragState.visStartY} x2={dragState.visPhase2StartX} y2={dragState.visPhase2StartY} stroke="white" strokeWidth="0.8" strokeDasharray="1,1" opacity="0.6" />
                    </g>
                  )}

                  {dragState.phase === 'position' ? (
                    <g>
                      <circle cx={dragState.visStartX} cy={dragState.visStartY} r="3" fill="#1e293b" />
                      <line x1={dragState.visStartX - 4} y1={dragState.visStartY} x2={dragState.visStartX - 14} y2={dragState.visStartY} stroke="#1e293b" strokeWidth="1.5" strokeLinecap="round" />
                      <line x1={dragState.visStartX + 4} y1={dragState.visStartY} x2={dragState.visStartX + 14} y2={dragState.visStartY} stroke="#1e293b" strokeWidth="1.5" strokeLinecap="round" />

                      <g transform={`translate(${dragState.visStartX - 22}, ${dragState.visStartY})`}>
                        <circle r="8" fill={dragState.visCurrentX - dragState.visStartX < -12 ? (activeTeam === 'A' ? '#4f46e5' : '#e11d48') : '#1e293b'} stroke="white" strokeWidth="1" />
                        <text x="0" y="2" fontSize="6" fontWeight="bold" fill="white" textAnchor="middle">GS</text>
                      </g>

                      <g transform={`translate(${dragState.visStartX + 22}, ${dragState.visStartY})`}>
                        <circle r="8" fill={dragState.visCurrentX - dragState.visStartX > 12 ? (activeTeam === 'A' ? '#4f46e5' : '#e11d48') : '#1e293b'} stroke="white" strokeWidth="1" />
                        <text x="0" y="2" fontSize="6" fontWeight="bold" fill="white" textAnchor="middle">GA</text>
                      </g>
                    </g>
                  ) : (
                    <g>
                      <circle cx={dragState.visPhase2StartX} cy={dragState.visPhase2StartY} r="3" fill="#1e293b" />
                      <line x1={dragState.visPhase2StartX} y1={dragState.visPhase2StartY - 4} x2={dragState.visPhase2StartX} y2={dragState.visPhase2StartY - 14} stroke="#1e293b" strokeWidth="1.5" strokeLinecap="round" />
                      <line x1={dragState.visPhase2StartX} y1={dragState.visPhase2StartY + 4} x2={dragState.visPhase2StartX} y2={dragState.visPhase2StartY + 14} stroke="#1e293b" strokeWidth="1.5" strokeLinecap="round" />

                      <g transform={`translate(${dragState.visPhase2StartX}, ${dragState.visPhase2StartY - 22})`}>
                        <circle r="8" fill={dragState.selectedResult === 'made' ? '#10b981' : '#1e293b'} stroke="white" strokeWidth="1" />
                        <text x="0" y="2.5" fontSize="7" fontWeight="bold" fill="white" textAnchor="middle">✓</text>
                      </g>

                      <g transform={`translate(${dragState.visPhase2StartX}, ${dragState.visPhase2StartY + 22})`}>
                        <circle r="8" fill={dragState.selectedResult === 'missed' ? '#ef4444' : '#1e293b'} stroke="white" strokeWidth="1" />
                        <text x="0" y="2.5" fontSize="7" fontWeight="bold" fill="white" textAnchor="middle">✗</text>
                      </g>
                    </g>
                  )}
                </g>
              )}

              {/* Hover Prediction Tooltip */}
              {hoverInfo && !dragState && (
                <g transform={`translate(${toVisCoords(hoverInfo.x, hoverInfo.y).x}, ${toVisCoords(hoverInfo.x, hoverInfo.y).y - 14})`} className="pointer-events-none transition-transform duration-75">
                  <rect x="-20" y="-14" width="40" height="14" rx="4" fill={hoverInfo.valid ? "#1e293b" : "#dc2626"} opacity="0.9" />
                  <text x="0" y="-4" fontSize="7" fontWeight="bold" fill="white" textAnchor="middle">
                    {hoverInfo.type === 'shot' ? (hoverInfo.valid ? `+${hoverInfo.points} PTS` : 'INVALID') : hoverInfo.type.toUpperCase()}
                  </text>
                  <circle cx="0" cy="14" r="4" fill="none" stroke={hoverInfo.valid ? "rgba(255,255,255,0.8)" : "rgba(220,38,38,0.8)"} strokeWidth="1" strokeDasharray="1,1" />
                </g>
              )}
            </svg>
          </div>
        </div>

        {/* Right Column: Event Log */}
        <div className="w-full lg:w-72 bg-white rounded-2xl shadow-sm border border-slate-200 flex flex-col h-[500px]">
          <div className="p-4 border-b border-slate-100 flex justify-between items-center bg-slate-50/50 rounded-t-2xl">
            <h2 className="text-sm font-bold text-slate-800 flex items-center gap-2">
              <BarChart2 className="w-4 h-4 text-blue-600" /> Play-by-Play
            </h2>
            <div className="flex gap-2">
              <button onClick={undoLastEvent} disabled={events.length === 0} className="p-1.5 text-slate-400 hover:text-orange-500 hover:bg-orange-50 rounded-lg transition-colors disabled:opacity-30" title="Undo Last Event">
                <Undo2 className="w-4 h-4" />
              </button>
              <button onClick={clearCourt} disabled={events.length === 0} className="p-1.5 text-slate-400 hover:text-red-500 hover:bg-red-50 rounded-lg transition-colors disabled:opacity-30" title="Clear All">
                <RotateCcw className="w-4 h-4" />
              </button>
            </div>
          </div>
          
          <div className="flex-1 overflow-y-auto p-3">
            {events.length === 0 ? (
              <div className="h-full flex flex-col items-center justify-center text-slate-400 text-sm p-6 text-center gap-3">
                <Target className="w-8 h-8 opacity-20" />
                <p>Log shots and stats on the court to see the play-by-play appear here.</p>
              </div>
            ) : (
              <ul className="space-y-2">
                {[...events].reverse().map((ev) => (
                  <li key={ev.id} className={`p-3 rounded-xl text-sm border-l-4 flex justify-between items-center shadow-sm ${
                    ev.team === 'A' ? 'bg-indigo-50/50 border-indigo-500' : 'bg-rose-50/50 border-rose-500'
                  }`}>
                    <div className="flex flex-col">
                      <span className={`font-black text-xs uppercase tracking-wider ${ev.team === 'A' ? 'text-indigo-600' : 'text-rose-600'}`}>
                        Team {ev.team} 
                        {ev.position && <span className="text-slate-500 bg-white px-1.5 rounded-sm border border-slate-200 ml-1">{ev.position}</span>}
                      </span>
                      <span className="text-slate-600 font-medium mt-0.5 capitalize">
                        {ev.type === 'shot' ? (ev.result === 'made' ? `Scored ${ev.points} PT` : `Missed ${ev.potentialPoints} PT attempt`) : ev.type}
                      </span>
                    </div>
                    {ev.type === 'shot' && ev.result === 'made' && <div className="bg-emerald-100 p-1.5 rounded-full"><CheckCircle2 className="w-4 h-4 text-emerald-600" /></div>}
                    {ev.type === 'shot' && ev.result === 'missed' && <div className="bg-rose-100 p-1.5 rounded-full"><XCircle className="w-4 h-4 text-rose-500" /></div>}
                    {ev.type === 'rebound' && <ArrowUpCircle className="w-5 h-5 text-purple-500" />}
                    {ev.type === 'turnover' && <RefreshCw className="w-5 h-5 text-orange-500" />}
                    {ev.type === 'intercept' && <Magnet className="w-5 h-5 text-cyan-500" />}
                    {ev.type === 'deflection' && <Hand className="w-5 h-5 text-teal-500" />}
                    {ev.type === 'penalty' && <AlertOctagon className="w-5 h-5 text-yellow-500" />}
                  </li>
                ))}
              </ul>
            )}
          </div>
        </div>

      </div>
    </div>
  );
}
